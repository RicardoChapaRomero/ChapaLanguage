PROGRAM
	DIM a, b, c, d, e, f, x, y, z AS INT

	GOSUB operaciones

	SUBPROCEDURE operaciones:
		let a = 0
		let b = 1
		let c = 2
		let d = 3
		let e = 4
		let f = 5
		let x = 6
		let y = 7
		let z = 8

		IF (a > b) THEN
			IF (b < c) THEN LET a = b
			ELSE LET b = c + d
			EIF
		ELSE
			IF (d > e) THEN LET d = f
			ELSE LET d = e
			EIF
		EIF

		WHILE (x < y) DO
			IF (a > b) THEN
				REPEAT
					INPUT "write", x
					LET y = 3
					LET z = 4
				UNTIL (x > a)
			ELSE
				WHILE (y < b) DO
					LET y = y + 1
				WEND
			EIF
		WEND

		RETURN
END
